import React, { useEffect, useState } from "react";
import { DisplayImages } from "./display-samples.mdx";

export const algo_cid = "aiverify_digital_corruptions";
import {
  brightness_down_images,
  brightness_up_images,
  contrast_down_images,
  contrast_up_images,
  saturate_images,
  random_tilt_images,
  compression_images,
} from "./digital-corruption-images.js";

export const TiltSamples = (props) => {
  // Create an array of URLs for the different severity levels (0-5)
  const tiltUrls = [];
  
  // Try to get images from artifacts first
  for (let i = 0; i <= 5; i++) {
    const artifactURL = props.getArtifactURL(
      algo_cid,
      `temp/Random_Tilt/severity${i}/9.png`,
      null
    );
    
    // If URL exists, add it to the array, otherwise use default images
    if (artifactURL) {
      tiltUrls.push(artifactURL);
    }
  }
  
  // If no artifacts were found, use the default images
  const imagesToUse = tiltUrls.length === 6 ? tiltUrls : random_tilt_images;

  return (
    <DisplayImages
      id={props.id}
      images={imagesToUse}
      data={props.getResults(algo_cid).results}
      corruption_index={5}
      corruption_name="Digital: Tilt"
    />
  );
}

<TiltSamples {...props} />
