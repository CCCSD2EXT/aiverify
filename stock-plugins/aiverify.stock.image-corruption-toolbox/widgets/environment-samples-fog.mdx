import React, { useEffect, useState } from "react";
import { DisplayImages } from "./display-samples.mdx";

export const algo_cid = "aiverify_environment_corruptions";
import {
  fog_images,
  snow_images,
  rain_images,
} from "./environment-corruption-images.js";

export const FogSamples = (props) => {
  // Create an array of URLs for the different severity levels (0-5)
  const fogUrls = [];
  
  // Try to get images from artifacts first
  for (let i = 0; i <= 5; i++) {
    const artifactURL = props.getArtifactURL(
      algo_cid,
      `temp/Fog/severity${i}/9.png`,
      null
    );
    
    // If URL exists, add it to the array, otherwise use default images
    if (artifactURL) {
      fogUrls.push(artifactURL);
    }
  }
  
  // If no artifacts were found, use the default images
  const imagesToUse = fogUrls.length === 6 ? fogUrls : fog_images;

  return (
    <DisplayImages
      id={props.id}
      images={imagesToUse}
      data={props.getResults(algo_cid).results}
      corruption_index={0}
      corruption_name="Environment: Fog"
    />
  );
}

<FogSamples {...props} />
